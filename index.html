PacketSync - game service packet synchronization for JS clients

Getting started:

require(
	[
		"http://spiderworm.github.io/packetsync/TimeDataStore.js",
		"http://spiderworm.github.io/packetsync/Timeline.js"
	],
	function(
		TimeDataStore,
		Timeline
	) {


		var store = new TimeDataStore(startTimestamp, startData);

		var timeline = new Timeline(id, store, startTimestamp);

		/*

			Now, every time you receive updates from the service,
			add those updates to the store.

		*/

		store.addUpdate(timestamp, update);


		/*

			The timeline will automatically start playing from the
			startTimestamp you provided it.  You should subscribe
			to updates from the timeline:

		*/

		timeline.onNewData(function(data) {
			// new data will be passed every animation frame
			// handle new data here
		});

	}
);
